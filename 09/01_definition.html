Generates solver definition file

Generates solver definition file (definition.xml) based on the information that BMI models provides about I/O.

Listing 45 shows the code to build definition.xml for CEM.

Listing 45 Code to build definition.xml for CEM
1import iricmi_pymt_util
2import pymt.models
3
4cem = pymt.models.Cem()
5iricmi_pymt_util.generateDefinition(cem, 'output/cem/definition.xml')

The code in Listing 45 outputs the code in Listing 46 to file ‘output/cem/definition.xml’. Please note that spaces and newline characters are added for readability.

Source code of gendef.py that generates definition.xml is shown in Listing 47.

Listing 47 source code of gendef.py
  1import datetime
  2import xml.etree.ElementTree as ET
  3
  4def generateDefinition(model, fname):
  5    root = ET.Element('SolverDefinition')
  6    _buildRootAttrs(model, root)
  7
  8    cc = ET.SubElement(root, 'CalculationCondition')
  9    tab = ET.SubElement(cc, 'Tab')
 10    tab.attrib['name'] = 'basic'
 11    tab.attrib['caption'] = 'Basic Settings'
 12    _buildCCs(model, tab)
 13
 14    grc = ET.SubElement(root, 'GridRelatedCondition')
 15    _buildGRCs(model, grc)
 16
 17    go = ET.SubElement(root, 'GlobalOutput')
 18    _buildGOs(model, go)
 19
 20    tree = ET.ElementTree(root)
 21    with open(fname, 'wb') as f:
 22        tree.write(f)
 23
 24def _buildRootAttrs(model, root):
 25    now = datetime.datetime.now()
 26
 27    root.attrib['name'] = model.get_component_name()
 28    root.attrib['caption'] = model.get_component_name()
 29    root.attrib['version'] = '1.0'
 30    root.attrib['copyright'] ="Please input value"
 31    root.attrib['release'] = now.strftime("%Y.%m.%d")
 32    root.attrib['homepage'] = "Please input value"
 33
 34    root.attrib['iterationtype'] = "time"
 35    root.attrib['gridtype'] ="structured2d"
 36
 37def _buildCCs(model, tab):
 38    for var_name in model.get_input_var_names():
 39        grid_id = model.get_var_grid(var_name)
 40        gt = model.get_grid_type(grid_id)
 41        if gt != 'scalar': continue
 42
 43        item = ET.SubElement(tab, 'Item')
 44        item.attrib['name'] = var_name
 45        item.attrib['caption'] = var_name + '[' + model.get_var_units(var_name) + ']'
 46        definition = ET.SubElement(item, 'Definition')
 47        definition.attrib['conditionType'] = 'functional'
 48        param = ET.SubElement(definition, 'Parameter')
 49        param.attrib['valueType'] = 'real'
 50        param.attrib['caption'] = 'Time'
 51        value = ET.SubElement(definition, 'Value')
 52        vt = model.get_var_type(var_name)
 53        if 'float' in vt:
 54            valueType = 'real'
 55        else:
 56            valueType = 'integer'
 57
 58        value.attrib['valueType'] = valueType
 59        value.attrib['caption'] = 'Value'
 60
 61def _buildGRCs(model, grc):
 62    for var_name in model.get_input_var_names():
 63        grid_id = model.get_var_grid(var_name)
 64        gt = model.get_grid_type(grid_id)
 65        if gt == 'scalar': continue
 66
 67        item = ET.SubElement(grc, 'Item')
 68        item.attrib['name'] = var_name
 69        item.attrib['caption'] = var_name + '[' + model.get_var_units(var_name) + ']'
 70
 71        vt = model.get_var_type(var_name)
 72        if 'float' in vt:
 73            valueType = 'real'
 74        else:
 75            valueType = 'integer'
 76
 77        definition = ET.SubElement(item, 'Definition')
 78        definition.attrib['valueType'] = valueType
 79        definition.attrib['position'] = model.get_var_location(var_name)
 80
 81        dim = ET.SubElement(definition, 'Dimension')
 82        dim.attrib['name'] = 'Time'
 83        dim.attrib['caption'] = 'Time'
 84        dim.attrib['valueType'] = 'real'
 85
 86    for var_name in model.get_output_var_names():
 87        grid_id = model.get_var_grid(var_name)
 88        gt = model.get_grid_type(grid_id)
 89        if gt == 'scalar': continue
 90
 91        output = ET.SubElement(grc, 'Output')
 92        output.attrib['name'] = var_name
 93        output.attrib['caption'] = var_name + '[' + model.get_var_units(var_name) + ']'
 94
 95        vt = model.get_var_type(var_name)
 96        if 'float' in vt:
 97            valueType = 'real'
 98        else:
 99            valueType = 'integer'
100
101        definition = ET.SubElement(output, 'Definition')
102        definition.attrib['valueType'] = valueType
103        definition.attrib['position'] = model.get_var_location(var_name)
104
105def _buildGOs(model, go):
106    for var_name in model.get_output_var_names():
107        grid_id = model.get_var_grid(var_name)
108        gt = model.get_grid_type(grid_id)
109        if gt != 'scalar': continue
110
111        output = ET.SubElement(go, 'Output')
112        output.attrib['name'] = var_name
113        output.attrib['caption'] = var_name + '[' + model.get_var_units(var_name) + ']'
114        definition = ET.SubElement(output, 'Definition')
115        vt = model.get_var_type(var_name)
116        if 'float' in vt:
117            valueType = 'real'
118        else:
119            valueType = 'integer'
120
121        definition.attrib['valueType'] = valueType
